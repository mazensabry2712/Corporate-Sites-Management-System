خطوات التثبيت
  • أضف ملف CSS في قسم @section('css')
    <link href="{{ URL::asset('assets/css/export-buttons.css') }}" rel="stylesheet">
  • أضف ملف JavaScript في قسم @section('js')
    <script src="{{ URL::asset('assets/js/export-functions.js') }}"></script>
  • أضف الأزرار في HTML الصفحة
    <div class="btn-group export-buttons"> <button class="btn btn-pdf" onclick="exportTableToPDF('example1', 'تقرير', [0, 5])"> <i class="fas fa-file-pdf"></i> </button> <!-- باقي الأزرار... --> </div>
  • نظف الذاكرة المؤقتة
    php artisan view:clear
المقارنة: القديم مقابل الجديد

JavaScript دوال مختلفة لكل صفحة مكتبة موحدة قابلة لإعادة الاستخدام
التصميم أشكال مختلفة وغير متسقة تصميم موحد احترافي
الصيانة تعديل 30+ ملف تعديل ملف واحد فقط
المكتبات تحميل يدوي في كل صفحة تحميل تلقائي عند الحاجة
معالجة الأخطاء غير موجودة أو أساسية معالجة شاملة مع رسائل واضحة
أمثلة سريعة
تصدير PDF
// تصدير جميع الأعمدة exportTableToPDF('example1', 'تقرير كامل', []) // استبعاد العمود الأول والأخير exportTableToPDF('example1', 'تقرير مفلتر', [0, 6]) // اسم ملف مخصص exportTableToPDF('example1', 'تقرير', [0, 6], 'report_2025.pdf')
تصدير Excel
// تصدير بيانات العملاء exportTableToExcel('customersTable', 'قائمة العملاء', [0, 5]) // تصدير مع اسم مخصص exportTableToExcel('vendorsTable', 'الموردين', [], 'vendors_oct_2025.xlsx')
طباعة
// طباعة تقرير المشاريع printTableData('projectsTable', 'تقرير المشاريع الشهري') // طباعة الفواتير printTableData('invoicesTable', 'قائمة الفواتير')
دليل المعاملات
المعامل النوع مطلوب؟ الوصف مثال
tableId نص ✅ نعم معرف الجدول في HTML 'example1'
title نص ✅ نعم عنوان التقرير 'تقرير المبيعات'
excludeColumns مصفوفة ❌ لا أرقام الأعمدة المستبعدة [0, 5, 6]
fileName نص ❌ لا اسم الملف المخصص 'report.pdf'
